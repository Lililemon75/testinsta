    <script>

        const databaseId = "https://www.notion.so/Test-widget-156bacdb4b9880ff808af4fa7ad149d9"; // Mets ici l'ID de ta base Notion

        const apiKey = "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"; // Mets ici ta clé API Notion

        async function fetchNotionData() {

            const response = await fetch(`https://api.notion.com/v1/databases/${databaseId}/query`, {

                method: "POST",

                headers: {

                    "Authorization": `Bearer ${apiKey}`,

                    "Notion-Version": "2022-06-28",

                    "Content-Type": "application/json",

                },

            });

            const data = await response.json();

            return data.results.map(item => ({

                title: item.properties["Titre"].title[0].text.content,

                image: item.properties["Image"].files[0].file.url,

                date: item.properties["Date de publication"].date.start,

            }));

        }

        async function renderProfile() {

            const posts = await fetchNotionData();

            const gridContainer = document.querySelector(".instagram-grid");

            document.getElementById('profile-name').textContent = "Nom du Profil";

            document.getElementById('profile-bio').textContent = "Bio ici";

            document.getElementById('post-count').textContent = posts.length;

            posts.forEach(post => {

                const postElement = document.createElement("div");

                postElement.classList.add("instagram-post");

                postElement.innerHTML = `<img src="${post.image}" alt="${post.title}">`;

                gridContainer.appendChild(postElement);

            });

        }

        renderProfile();
